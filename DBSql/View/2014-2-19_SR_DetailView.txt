drop view [dbo].[SR_DetailView]

CREATE  VIEW dbo.SR_DetailView
AS
SELECT dbo.SR_Detail.GUID, dbo.SR_Detail.GUID_SR_Main, dbo.SR_Detail.OAOTS, 
      dbo.SR_Detail.GUID_SRType, dbo.SS_SRType.SRTypeKey, 
      dbo.SS_SRType.SRTypeName, dbo.SR_Detail.OrderNum, dbo.SR_Detail.ActionDate, 
      dbo.SR_Detail.GUID_BGCode, dbo.SS_BGCode.BGCodeKey, 
      dbo.SS_BGCode.BGCodeName, dbo.SR_Detail.GUID_Department, 
      dbo.SS_Department.DepartmentKey, dbo.SS_Department.DepartmentName, 
      dbo.SR_Detail.GUID_Person, dbo.SS_Person.PersonKey, 
      dbo.SS_Person.PersonName, dbo.SR_Detail.GUID_ProjectKey, 
      dbo.SS_Project.ProjectKey, dbo.SS_Project.ProjectName, 
      dbo.SS_Project.GUID_ProjectClass,dbo.SS_ProjectClass.ProjectClassKey,dbo.SS_ProjectClass.ProjectClassName,
      dbo.SR_Detail.GUID_Cutomer, dbo.SS_Customer.CustomerKey, 
      dbo.SS_Customer.CustomerName, dbo.SR_Detail.GUID_SettleType, 
      dbo.SS_SettleType.SettleTypeKey, dbo.SS_SettleType.SettleTypeName, 
      dbo.SR_Detail.ActionMemo, dbo.SR_Detail.Total_SR, dbo.SR_Detail.Total_Tax, 
      dbo.SR_Detail.Total_Real, dbo.SR_Detail.MakeDate, dbo.SR_Detail.ModifyDate, 
      dbo.SR_Detail.GUID_Maker, SS_Operator_1.OperatorName AS Maker, 
      dbo.SR_Detail.GUID_Modifier, SS_Operator_1.OperatorName AS Modifier, 
      dbo.SR_Detail.GUID_PaymentNumber,dbo.SR_Detail.GUID_BGType, dbo.BG_Type.BGTypeKey, 
      dbo.BG_Type.BGTypeName
FROM dbo.SR_Detail LEFT OUTER JOIN
      dbo.SS_Operator SS_Operator_1 ON 
      dbo.SR_Detail.GUID_Modifier = SS_Operator_1.GUID LEFT OUTER JOIN
      dbo.SS_Operator SS_Operator_2 ON 
      dbo.SR_Detail.GUID_Maker = SS_Operator_2.GUID LEFT OUTER JOIN
      dbo.SS_SettleType ON 
      dbo.SR_Detail.GUID_SettleType = dbo.SS_SettleType.GUID LEFT OUTER JOIN
      dbo.SS_Customer ON 
      dbo.SR_Detail.GUID_Cutomer = dbo.SS_Customer.GUID LEFT OUTER JOIN
      dbo.SS_Project ON 
      dbo.SR_Detail.GUID_ProjectKey = dbo.SS_Project.GUID LEFT OUTER JOIN
      dbo.SS_ProjectClass ON
      dbo.SS_Project.GUID_ProjectClass=dbo.SS_ProjectClass.GUID LEFT OUTER JOIN
      dbo.SS_Person ON 
      dbo.SR_Detail.GUID_Person = dbo.SS_Person.GUID LEFT OUTER JOIN
      dbo.SS_Department ON 
      dbo.SR_Detail.GUID_Department = dbo.SS_Department.GUID LEFT OUTER JOIN
      dbo.SS_BGCode ON 
      dbo.SR_Detail.GUID_BGCode = dbo.SS_BGCode.GUID LEFT OUTER JOIN
      dbo.SS_SRType ON dbo.SR_Detail.GUID_SRType = dbo.SS_SRType.GUID LEFT OUTER JOIN
	  dbo.BG_Type ON 
      dbo.SR_Detail.GUID_BGType = dbo.BG_Type.GUID 


GO
SET QUOTED_IDENTIFIER OFF 
GO
SET ANSI_NULLS ON 
GO
